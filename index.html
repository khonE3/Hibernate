    <property name="hibernate.hbm2ddl.auto">update</property>
    <property name="hibernate.show_sql">true</property>
    <property name="hibernate.format_sql">true</property>

SessionFactory เป็น thread-safe object สร้างครั้งเดียวตลอด application lifecycle

Java
import org.hibernate.SessionFactory;
import org.hibernate.cfg.Configuration;

public class HibernateUtil {
    private static final SessionFactory sessionFactory;

    static {
        try {
            sessionFactory = new Configuration()
                .configure("hibernate.cfg.xml")
                .buildSessionFactory();
        } catch (Throwable ex) {
            throw new ExceptionInInitializerError(ex);
        }
    }

    public static SessionFactory getSessionFactory() {
        return sessionFactory;
    }
}

Entity Mapping Annotations

Annotations สำหรับ map Java class ไปยัง database table

@Entity / @Tableประกาศ entity class และกำหนดชื่อตาราง
JPAสำคัญ
▼
Java
import javax.persistence.*;

@Entity
@Table(name = "employees",
       uniqueConstraints = @UniqueConstraint(columnNames = {"email"}))
public class Employee {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "full_name", nullable = false, length = 100)
    private String name;

    @Column(unique = true, length = 150)
    private String email;

    @Column(precision = 10, scale = 2)
    private Double salary;

    public Employee() {}
}
💡
Entity class ต้องมี no-arg constructor (public หรือ protected)
@Id / @GeneratedValueกำหนด Primary Key และ strategy การสร้าง ID
JPA
▼
Java
// Strategy 1: IDENTITY — auto-increment ของ DB
@Id
@GeneratedValue(strategy = GenerationType.IDENTITY)
private Long id;

// Strategy 2: SEQUENCE — ใช้ sequence ของ DB
@Id
@GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "emp_seq")
@SequenceGenerator(name = "emp_seq", sequenceName = "employee_sequence", allocationSize = 1)
private Long id;

// Strategy 3: AUTO — Hibernate เลือกให้อัตโนมัติ
@Id
@GeneratedValue(strategy = GenerationType.AUTO)
private Long id;
@Columnกำหนดรายละเอียดของ column
JPA
▼
Java
@Column(
    name = "full_name",         // ชื่อ column ใน DB
    nullable = false,            // NOT NULL
    unique = true,               // UNIQUE
    length = 100,                // VARCHAR length
    insertable = true,           // รวมใน INSERT
    updatable = true,            // รวมใน UPDATE
    columnDefinition = "TEXT",   // กำหนด SQL type
    precision = 10,              // decimal ตัวเลขทั้งหมด
    scale = 2                     // decimal ทศนิยม
)
private String name;
@Transient / @Lob / @Enumerated / @TemporalAnnotations สำหรับ field พิเศษ
JPA
▼
Java
@Entity
public class Product {
    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Transient                          // ไม่ถูกเก็บใน DB
    private String temporaryNote;

    @Lob                                // Large Object (CLOB/BLOB)
    private String description;

    @Enumerated(EnumType.STRING)         // เก็บเป็นชื่อ: "ACTIVE"
    private Status status;

    @Temporal(TemporalType.TIMESTAMP)    // วันที่+เวลา
    private Date createdAt;
}
enum Status { ACTIVE, INACTIVE, PENDING }

Java

@DynamicInsert / @DynamicUpdate / @NaturalIdHibernate-specific annotations
Hibernate
▼
Java
import org.hibernate.annotations.*;

@Entity
@DynamicInsert     // INSERT เฉพาะ column ที่ไม่เป็น null
@DynamicUpdate     // UPDATE เฉพาะ column ที่เปลี่ยน
public class User {
    @Id @GeneratedValue
    private Long id;

    @NaturalId         // Business key — immutable, unique
    private String email;

    @CreationTimestamp // บันทึกเวลาตอนสร้าง
    private LocalDateTime createdAt;

    @UpdateTimestamp   // บันทึกเวลาตอนอัพเดท
    private LocalDateTime updatedAt;

    @Formula("salary * 12")  // คำนวณจาก SQL expression
    private Double annualSalary;
}

Session API

Session เป็น core interface ของ Hibernate สำหรับ CRUD operations

session.persist()บันทึก entity ใหม่ลง DB (INSERT)
JPAแนะนำ
▼

เปลี่ยนสถานะ entity จาก Transient → Persistent

Java
Session session = HibernateUtil.getSessionFactory().openSession();
Transaction tx = session.beginTransaction();

Employee emp = new Employee();
emp.setName("สมชาย ใจดี");
emp.setEmail("somchai@example.com");

session.persist(emp);   // Transient → Persistent
tx.commit();
session.close();
session.save()บันทึก entity + return ID ทันที
HibernateDeprecated
▼
Java
Serializable id = session.save(emp);  // return ID ทันที
System.out.println("Generated ID: " + id);
⚠️
save() deprecated ใน Hibernate 6+ ใช้ persist() แทน
session.get() / session.load()ดึง entity ด้วย Primary Key
Hibernate
▼
Java
// get() — query DB ทันที, return null ถ้าไม่เจอ
Employee emp1 = session.get(Employee.class, 1L);

// load() — return proxy (lazy), exception ถ้าไม่เจอ
Employee emp2 = session.load(Employee.class, 2L);

session.remove() / session.delete()ลบ entity จาก DB (DELETE)
JPA
▼
Java
Employee emp = session.get(Employee.class, 1L);
if (emp != null) {
    session.remove(emp);   // JPA standard (แนะนำ)
    // session.delete(emp); // Hibernate-specific (deprecated)
}
flush() / clear() / evict() / refresh()จัดการ Session cache
Hibernate
▼
Java
session.flush();     // sync changes กับ DB ทันที
session.refresh(emp); // โหลดข้อมูลจาก DB มาทับ
session.evict(emp);   // ลบ entity ออกจาก cache → detached
session.clear();      // ล้าง first-level cache ทั้งหมด

session.createQuery()สร้าง HQL/JPQL query

▼

    .setMaxResults(10)

Transaction Management

ทุกการเปลี่ยนแปลงข้อมูลต้องอยู่ภายใน Transaction

Transaction Patternรูปแบบมาตรฐาน try-catch-finally
สำคัญ
▼
Java
Session session = null;
Transaction tx = null;
try {
    session = HibernateUtil.getSessionFactory().openSession();
    tx = session.beginTransaction();

    // Business Logic
    Employee emp = new Employee("สมศรี", "somsri@example.com", 60000.0);
    session.persist(emp);

    tx.commit();       // สำเร็จ → commit
} catch (Exception e) {
    if (tx != null) tx.rollback();  // ผิดพลาด → rollback
    e.printStackTrace();
} finally {
    if (session != null) session.close();
}

HQL / JPQL Queries

Hibernate Query Language — ภาษาคล้าย SQL แต่อ้างอิงจาก Entity class

HQL SELECT / WHERE / JOINQuery ข้อมูลแบบต่างๆ
Query
▼
Java
// SELECT ทั้งหมด
List<Employee> all = session.createQuery("FROM Employee", Employee.class).getResultList();

// WHERE + Named Parameters
List<Employee> filtered = session
    .createQuery("FROM Employee e WHERE e.salary > :min AND e.name LIKE :name", Employee.class)
    .setParameter("min", 30000.0)
    .setParameter("name", "%สม%")
    .getResultList();

// JOIN FETCH (แก้ N+1 problem)
List<Department> depts = session
    .createQuery("FROM Department d JOIN FETCH d.employees", Department.class)
    .getResultList();

// Aggregate Functions
Long count = session.createQuery("SELECT COUNT(e) FROM Employee e", Long.class).uniqueResult();
Double avg = session.createQuery("SELECT AVG(e.salary) FROM Employee e", Double.class).uniqueResult();

// Bulk UPDATE / DELETE
int updated = session.createQuery("UPDATE Employee SET salary = salary * 1.1 WHERE salary < :min")
    .setParameter("min", 30000.0)
    .executeUpdate();

Criteria API

Type-safe query builder — ไม่ต้องเขียน HQL string ป้องกัน typo

CriteriaBuilder / CriteriaQueryสร้าง query แบบ Type-Safe
JPA
▼
Java
CriteriaBuilder cb = session.getCriteriaBuilder();
CriteriaQuery<Employee> cr = cb.createQuery(Employee.class);
Root<Employee> root = cr.from(Employee.class);

// Predicates
Predicate salaryGt = cb.greaterThan(root.get("salary"), 50000.0);
Predicate nameLike = cb.like(root.get("name"), "%สม%");

cr.select(root).where(cb.and(salaryGt, nameLike));
cr.orderBy(cb.desc(root.get("salary")));

List<Employee> results = session.createQuery(cr).getResultList();

// Aggregate: count, sum, avg, max, min
CriteriaQuery<Long> countQ = cb.createQuery(Long.class);
countQ.select(cb.count(countQ.from(Employee.class)));
Long total = session.createQuery(countQ).getSingleResult();

Relationship Mapping

การแมปความสัมพันธ์ระหว่าง Entity

@OneToOneความสัมพันธ์ 1:1
JPA
▼
Java
@Entity
public class User {
    @Id @GeneratedValue
    private Long id;

    @OneToOne(cascade = CascadeType.ALL)
    @JoinColumn(name = "profile_id")
    private UserProfile profile;
}

@Entity
public class UserProfile {
    @Id @GeneratedValue
    private Long id;

    @OneToOne(mappedBy = "profile")
    private User user;
}
@OneToMany / @ManyToOneความสัมพันธ์ 1:N
JPA
▼
Java
@Entity
public class Department {
    @Id @GeneratedValue
    private Long id;

    @OneToMany(mappedBy = "department", cascade = CascadeType.ALL, orphanRemoval = true)
    private List<Employee> employees = new ArrayList<>();
}

@Entity
public class Employee {
    @Id @GeneratedValue
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "dept_id")
    private Department department;
}
@ManyToManyความสัมพันธ์ N:N
JPA
▼
Java
@Entity
public class Student {
    @Id @GeneratedValue
    private Long id;

    @ManyToMany
    @JoinTable(name = "student_course",
        joinColumns = @JoinColumn(name = "student_id"),
        inverseJoinColumns = @JoinColumn(name = "course_id"))
    private Set<Course> courses = new HashSet<>();
}

@Entity
public class Course {
    @Id @GeneratedValue
    private Long id;

    @ManyToMany(mappedBy = "courses")
    private Set<Student> students = new HashSet<>();
}

Inheritance Mapping

แมปการสืบทอด Class ลงฐานข้อมูล

SINGLE_TABLEรวมทุก subclass ไว้ตารางเดียว (เร็วสุด)
JPADefault
▼
Java
@Entity
@Inheritance(strategy = InheritanceType.SINGLE_TABLE)
@DiscriminatorColumn(name = "vehicle_type")
public abstract class Vehicle {
    @Id @GeneratedValue
    private Long id;
    private String brand;
}

@Entity @DiscriminatorValue("CAR")
public class Car extends Vehicle {
    private int numberOfDoors;
}

@Entity @DiscriminatorValue("BIKE")
public class Bike extends Vehicle {
    private boolean hasBasket;
}
JOINEDแยกตารางแต่ละ class ใช้ FK โยงกัน
JPA
▼
Java
@Entity
@Inheritance(strategy = InheritanceType.JOINED)
public abstract class Animal {
    @Id @GeneratedValue
    private Long id;
    private String name;
}

@Entity
public class Dog extends Animal {
    private String breed;
}
TABLE_PER_CLASSแยกตารางแต่ละ subclass คัดลอก field ทั้งหมด
JPAไม่แนะนำ
▼
Java
@Entity
@Inheritance(strategy = InheritanceType.TABLE_PER_CLASS)
public abstract class BillingDetails {
    @Id private Long id;
    private String owner;
}

@Entity
public class CreditCard extends BillingDetails {
    private String cardNumber;
}

ไม่สามารถใช้ @GeneratedValue(IDENTITY) ได้ และ polymorphic query ช้ามากเนื่องจาก UNION

Entity Lifecycle Callbacks

ดัก Event เวลา Entity เปลี่ยนสถานะ

@PrePersist / @PostPersist / @PreUpdate / @PostLoadCallback methods
JPA
▼
Java
@Entity
public class Article {
    @Id @GeneratedValue
    private Long id;
    private Date createdAt;
    private Date updatedAt;

    @PrePersist
    protected void onCreate() {
        createdAt = new Date();
        updatedAt = new Date();
    }

    @PreUpdate
    protected void onUpdate() {
        updatedAt = new Date();
    }

    @PostPersist
    public void afterInsert() {
        System.out.println("Inserted ID: " + id);
    }

    @PostLoad
    public void afterLoad() {
        System.out.println("Loaded: " + id);
    }
}

Batch Processing

ลูป persist ทีละแสน record ต้อง flush + clear เป็นระยะ ไม่งั้น Out Of Memory

Batch Insert / UpdateJDBC batch + flush clearing
Best Practice
▼

ตั้งค่า:

XML
<property name="hibernate.jdbc.batch_size">50</property>
<property name="hibernate.order_inserts">true</property>
<property name="hibernate.order_updates">true</property>

โค้ด Java:

Java
Session session = HibernateUtil.getSessionFactory().openSession();
Transaction tx = session.beginTransaction();

for (int i = 0; i < 100000; i++) {
    Employee emp = new Employee("Employee " + i);
    session.persist(emp);

    // ทุกๆ 50 records → flush & clear
    if (i > 0 && i % 50 == 0) {
        session.flush();
        session.clear();
    }
}

tx.commit();
session.close();
💡
batch_size ที่เหมาะสมคือ 25-50 ขึ้นอยู่กับขนาดข้อมูลแต่ละ row
